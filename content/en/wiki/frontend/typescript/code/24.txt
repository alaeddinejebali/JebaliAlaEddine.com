class Person{
	private _name;
	constructor(){
	}
	get name(){
		return this._name;
	}
	set name(name){
		console.log('constructor called...');
		if(name === 'anonymous'){
			this._name = 'not registered user';
		}else{
			this._name = name;
		}
	}
}

var p:Person = new Person();
p.name = "Ala";
console.log(p.name);
//outputs: 
//constructor called...
//Ala

p.name = "anonymous";
console.log(p.name);
//outputs: 
//constructor called...
//anonymous
